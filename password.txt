        // عناصر DOM

        let autoHideTimeout = null;

        checkStrengthBtn.addEventListener('click', checkPasswordStrength);
        checkBreachBtn.addEventListener('click', checkPasswordBreach);
        generatePasswordBtn.addEventListener('click', generateStrongPassword);

        // وظائف
        function checkPasswordStrength() {
            const password = passwordInput.value;
            
            if (!password) {
                alert('يرجى إدخال كلمة مرور للفحص');
                return;
            }
            
            // تحليل قوة كلمة المرور
            const analysis = analyzePassword(password);
            updateStrengthUI(analysis);
        }
        
        function analyzePassword(password) {
            let score = 0;
            const analysis = {
                hasLowerCase: /[a-z]/.test(password),
                hasUpperCase: /[A-Z]/.test(password),
                hasNumbers: /\d/.test(password),
                hasSpecialChars: /[!@#$%^&*()_+\-=\[\]{};':"\\|,.<>\/?]/.test(password),
                isLongEnough: password.length >= 12,
                length: password.length
            };
            
            // حساب النقاط
            if (analysis.hasLowerCase) score += 1;
            if (analysis.hasUpperCase) score += 1;
            if (analysis.hasNumbers) score += 1;
            if (analysis.hasSpecialChars) score += 1;
            if (analysis.isLongEnough) score += 2;
            
            // حساب النسبة المئوية
            const maxScore = 6;
            const percentage = (score / maxScore) * 100;
            
            return {
                ...analysis,
                score: percentage
            };
        }
        
        function updateStrengthUI(analysis) {
            // تحديث شريط القوة
            strengthFill.style.width = `${analysis.score}%`;
            
            // تحديث اللون والنص
            let strengthLevel, color, bgColor;
            
            if (analysis.score < 40) {
                strengthLevel = 'ضعيفة';
                color = 'var(--danger-color)';
                bgColor = 'rgba(231, 76, 60, 0.1)';
                strengthFill.style.background = 'linear-gradient(to right, var(--danger-color), var(--danger-dark))';
            } else if (analysis.score < 70) {
                strengthLevel = 'متوسطة';
                color = 'var(--warning-color)';
                bgColor = 'rgba(243, 156, 18, 0.1)';
                strengthFill.style.background = 'linear-gradient(to right, var(--warning-color), var(--warning-dark))';
            } else {
                strengthLevel = 'قوية';
                color = 'var(--success-color)';
                bgColor = 'rgba(46, 204, 113, 0.1)';
                strengthFill.style.background = 'linear-gradient(to right, var(--success-color), var(--success-dark))';
            }
            
            strengthText.textContent = `قوة كلمة المرور: ${strengthLevel}`;
            strengthText.style.color = color;
            strengthText.style.background = bgColor;
            
            // تحديث التفاصيل
            strengthDetails.innerHTML = '';
            
            const details = [
                { condition: analysis.hasLowerCase, text: 'تحتوي على أحرف صغيرة', icon: analysis.hasLowerCase ? 'fas fa-check-circle' : 'fas fa-times-circle' },
                { condition: analysis.hasUpperCase, text: 'تحتوي على أحرف كبيرة', icon: analysis.hasUpperCase ? 'fas fa-check-circle' : 'fas fa-times-circle' },
                { condition: analysis.hasNumbers, text: 'تحتوي على أرقام', icon: analysis.hasNumbers ? 'fas fa-check-circle' : 'fas fa-times-circle' },
                { condition: analysis.hasSpecialChars, text: 'تحتوي على رموز خاصة', icon: analysis.hasSpecialChars ? 'fas fa-check-circle' : 'fas fa-times-circle' },
                { condition: analysis.isLongEnough, text: 'الطول مناسب (12 حرفًا على الأقل)', icon: analysis.isLongEnough ? 'fas fa-check-circle' : 'fas fa-times-circle' },
                { condition: analysis.length > 0, text: `الطول: ${analysis.length} حرف`, icon: 'fas fa-ruler' }
            ];
            
            details.forEach(detail => {
                const item = document.createElement('div');
                item.className = 'strength-item';
                item.innerHTML = `
                    <i class="${detail.icon}" style="color: ${detail.condition ? 'var(--success-color)' : 'var(--danger-color)'}"></i>
                    <span>${detail.text}</span>
                `;
                strengthDetails.appendChild(item);
            });
        }
        
        function togglePasswordVisibility() {
            const type = passwordInput.getAttribute('type') === 'password' ? 'text' : 'password';
            passwordInput.setAttribute('type', type);
            
            // تغيير الأيقونة
            togglePassword.innerHTML = type === 'password' ? '<i class="fas fa-eye"></i>' : '<i class="fas fa-eye-slash"></i>';
            
            // إذا تم إظهار كلمة المرور، قم بإخفائها تلقائياً بعد 3 ثوانٍ
            if (type === 'text') {
                // إلغاء أي مؤقت سابق
                if (autoHideTimeout) {
                    clearTimeout(autoHideTimeout);
                }
                
                // تعيين مؤقت جديد لإخفاء كلمة المرور بعد 3 ثوانٍ
                autoHideTimeout = setTimeout(() => {
                    passwordInput.setAttribute('type', 'password');
                    togglePassword.innerHTML = '<i class="fas fa-eye"></i>';
                }, 3000);
            } else {
                // إذا تم إخفاء كلمة المرور، إلغاء المؤقت
                if (autoHideTimeout) {
                    clearTimeout(autoHideTimeout);
                }
            }
        }
        
        function checkPasswordBreach() {
            const password = passwordInput.value;
            
            if (!password) {
                alert('يرجى إدخال كلمة مرور للبحث عن التسريبات');
                return;
            }
            
            // محاكاة البحث في قاعدة البيانات المسربة

            breachResult.style.display = 'block';
            
            // محاكاة البحث (في التطبيق الحقيقي، سيتم استدعاء API هنا)
            setTimeout(() => {
                // محاكاة نتيجة البحث
                const isBreached = Math.random() < 0.3; // 30% فرصة أن تكون مسربة
                
                if (isBreached) {
                    breachResult.className = 'breach-result breach-found';
                    breachResult.innerHTML = '<i class="fas fa-exclamation-triangle"></i> تنبيه: تم العثور على كلمة المرور هذه في قاعدة البيانات المسربة! يوصى بتغييرها فوراً.';
                } else {
                    breachResult.className = 'breach-result breach-not-found';
                    breachResult.innerHTML = '<i class="fas fa-check-circle"></i> ممتاز! لم يتم العثور على كلمة المرور هذه في قاعدة البيانات المسربة.';
                }
            }, 1500);
        }
        
        function generateStrongPassword() {
            // توليد كلمة مرور قوية
            const lowercase = 'abcdefghijklmnopqrstuvwxyz';
            const uppercase = 'ABCDEFGHIJKLMNOPQRSTUVWXYZ';
            const numbers = '0123456789';
            const symbols = '!@#$%^&*()_+-=[]{};:,.<>?';
            
            let password = '';
            
            // التأكد من احتواء كلمة المرور على جميع الأنواع
            password += getRandomChar(lowercase);
            password += getRandomChar(uppercase);
            password += getRandomChar(numbers);
            password += getRandomChar(symbols);
            
            // إكمال كلمة المرور إلى 16 حرفًا
            const allChars = lowercase + uppercase + numbers + symbols;
            for (let i = password.length; i < 16; i++) {
                password += getRandomChar(allChars);
            }
            
            // خلط الأحرف عشوائيًا
            password = shuffleString(password);
            
            // عرض كلمة المرور المولدة
            currentGeneratedPassword = password;
            generatedPassword.innerHTML = `<i class="fas fa-key"></i> ${password}`;
            generatedPassword.style.display = 'block';
            actionButtons.style.display = 'flex';
        }
        
        function getRandomChar(characters) {
            return characters[Math.floor(Math.random() * characters.length)];
        }
        
        function shuffleString(string) {
            const array = string.split('');
            for (let i = array.length - 1; i > 0; i--) {
                const j = Math.floor(Math.random() * (i + 1));
                [array[i], array[j]] = [array[j], array[i]];
            }
            return array.join('');
        }
        
        function useGeneratedPassword() {
            passwordInput.value = currentGeneratedPassword;
            checkPasswordStrength();
            hideGeneratedPassword();
            alert('تم استخدام كلمة المرور المولدة. يمكنك الآن فحص قوتها.');
        }
        
        function hideGeneratedPassword() {
            generatedPassword.style.display = 'none';
            actionButtons.style.display = 'none';
        }
